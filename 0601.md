 ## 268. Missing Number
  - sort
 ```py
class Solution:
    def missingNumber(self, nums: List[int]) -> int:
        nums.sort()

        for i, num in enumerate(nums):
            if num != i:
                return i
        return len(nums)

        # time: O(nlogn)
        # space: O(n)       
 ```
 - set
 ```py
 class Solution:
    def missingNumber(self, nums: List[int]) -> int:
        num_set = set(nums)
        for i in range(len(nums) + 1):
            if i not in num_set:
                return i
                
        # time: O(n)
        # space: O(n)      
 ```
 - bit
 ```py
 class Solution:
    def missingNumber(self, nums: List[int]) -> int:
        res = len(nums)
        for i, num in enumerate(nums):
            res ^= i ^ num
        
        return res
        
        # time: O(n)
        # space: O(1)   
 ```
