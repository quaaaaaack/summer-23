class Solution:

        def compare(x, y):

## 2384. Largest Palindromic Number
```py
class Solution:
    def largestPalindromic(self, num: str) -> str:
        freqs = collections.Counter(num)
        # edge cases
        if freqs['0'] == len(num):
            return "0"

        # left side
        left = ""
        for i in "987654321":
            left += i * (freqs[i] // 2)
        if left:
            left += '0' * (freqs['0'] // 2)

        right = left[::-1]

        center = ""

        for i in "9876543210":
            if freqs[i] % 2 == 1:
                center = i
                break
        
        return left + center + right
```
